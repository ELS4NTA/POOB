public void flashCam(int x,int y, double angle) {
        this.isVisible = true;
        if(isVisible) {
            Canvas canvas = Canvas.getCanvas();
            this.angle = (int) angle;
            xPosition = (int) (x*Math.cos(this.angle)+((width/2)-y)*Math.sin(this.angle));
            yPosition = (int) (((width/2)-y)*Math.cos(this.angle)-x*Math.sin(this.angle));
            canvas.draw(this, color,new java.awt.Rectangle(xPosition, yPosition, 
                                       width, height),this.angle,x,y);
            canvas.wait(100);
            
        }
        makeInvisible();
    }
